
\section{Introduction}

\begin{frame}{Pourquoi les langages de script ?}
  \begin{itemize}
    \item Automatisation des tâches répétitives.
    \item Prototypage rapide.
    \item Glue entre différents outils/logiciels.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Exemple de script Bash}
  \begin{minted}{bash}
  #!/bin/bash
  echo "Bonjour, monde !"
  for i in {1..5}; do
    echo "Itération $i"
  done
  \end{minted}
\end{frame}
